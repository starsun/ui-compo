异步分页使用方式：
在js中引入下面的分页组件：	

ALP.Import('widget/widget.js');
ALP.Import('util/string.js');
ALP.Import('widget/page.js');


html中添加分页容器：

<div id="pagebar"></div>


通常在异步获取数据后调用create和init方法：

/**添加分页**/
$("#pagebar a").unbind();//卸载事件
var query = {curPage:4,totalPage:12};//从异步数据中获得分页信息	
Widget.Page.create("#pagebar", query);
Widget.Page.init("#pagebar", function(p){//回调函数，通常是点击分页后重新请求数据
	parame.curPage = p;
	self.getPage(parame);
});
